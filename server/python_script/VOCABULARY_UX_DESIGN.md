### 1. 資料庫查詢優化
```javascript
// 按主題查詢
db.book_words.find({"categories": "家庭生活"})

// 按難度查詢  
db.book_words.find({"learning_level": "beginner"})

// 複合查詢
db.book_words.find({
  "learning_level": "beginner",
  "categories": "日常動作",
  "frequency": "high"
})

// 全文搜尋
db.book_words.find({$text: {$search: "家庭"}})
```

